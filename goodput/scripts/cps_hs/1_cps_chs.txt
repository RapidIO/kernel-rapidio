// Script to run on controller in CPS hot swap test.
//
// This starts threads for receiving and handling
// Tsi721 and CPS port-writes, and also starts a
// thread for polling CPS status.
thread 0 -1 0
thread 1 -1 0
thread 2 -1 0
thread 14 -1 0
wait 0 H
IBA 0 0x100000 0x200000000
sleep 2
wait 0 H
wait 1 H
wait 2 H
wait 14 H
CH 0 T
wait 0 R
CH 1 C
wait 1 R
CH 2 P 0 1
wait 2 R
sleep 2
CH 14 R 1
wait 14 R
